Set up Serial Port

%Create serial port object
ComPortName = 'COM4';
ComPort = serial(ComPortName);

%Configure serial port
set(ComPort,'BaudRate',38400,'Parity','none');
ComPort. StopBit = 1;
ComPort.Terminator = 'CR';

fprintf('Current Serial Port Settings : \n');
get(ComPort);

%Open Serial Port
fopen(ComPort);
